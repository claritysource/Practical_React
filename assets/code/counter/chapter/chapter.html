The Solution
 
      
      
The code for the solution is shown in the following listing

      
1 import React, { useState } from "react"
2 import "./App.css"
3 
4 const Button = ({
5   style,
6   ...props
7 }) => (
8   <button
9     style={{
10       border: "none",
11       padding: "16px",
12       borderRadius: "4px",
13       background: "lightblue",
14       margin: "5px",
15       ...style,
16     }}
17     {...props}
18   />
19 )
20 
21 const Counter = ({
22   twoButtons = false
23 }) => {
24   const [count, setCount] = useState(0)
25 
26   const increment = () => setCount(prev => prev + 1)
27   const decrement = () => setCount(prev => prev - 1)
28 
29   return twoButtons ?
30     <div>
31       <Button
32         style={{ backgroundColor: "red" }}
33         onClick={increment}
34       >
35         increase
36       </Button>
37 
38       <Button
39          style={{ backgroundColor: "purple" }}
40          onClick={decrement}
41        >
42         decrease
43       </Button>
44 
45       {count}
46     </div>
47   :
48     <Button
49       onClick={() => setCount(prev => prev + 1)}>
50         {count}
51       </Button>
52 }
53 
54 const Solution = () => {
55   return (
56     <div>
57       <Counter />
58       <Counter twoButtons={true} />
59     </div>
60   )
61 }
62 
63 export default Solution
